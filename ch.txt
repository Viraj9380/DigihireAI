import jwt
import uuid
from datetime import datetime, timedelta
from app.core.config import settings

def generate_invite_token(email: str, candidate_id: str, assessment_candidate_id: str, company_id: str, expiry_days: int) -> tuple[str, str]:
    """
    Returns (jwt_token, jti).
    expiry_days is passed from caller (company-specific).
    """
    jti = str(uuid.uuid4())
    payload = {
    "sub": "candidate_invite",
    "email":str(email),
    "cid": str(candidate_id),
    "aci": str(assessment_candidate_id),
    "coid": str(company_id),   # âœ… add this
    "jti": jti,
    "exp": datetime.utcnow() + timedelta(days=expiry_days),
    "iat": datetime.utcnow(),
}

    token = jwt.encode(payload, settings.JWT_SECRET_KEY, algorithm=settings.JWT_ALGORITHM)
    return token, jti
